	id = magiks_event.4
	title = BLANK_STRING
	picture = GFX_evt_hyperlane
	desc = magiks_decision_hyperlane_found_desc
	is_triggered_only = yes
	option = {
		name = LETS_START
		custom_tooltip = magiks_event_hyperlane_in_border_resource_fail
		allow = {
			space_owner = {
				has_resource = {
					type = energy
					amount >= 1000
				}
				has_resource = {
					type = manacite
					amount >= 50
				}
				has_resource = {
					type = influence
					amount >= 1
				}
			}
		}
		hidden_effect = {
			log = "Core Planet: [From.GetName] Other System: [This.GetName]"

				from = From.solar_system
				to = This
			}
			space_owner = {
				add_resource = {
					energy = -1000
					manacite = -50
					influence = -1
				}
			}
		} 
	}
	option = {
		name = NEVERMIND
	}
}

	id = magiks_event.8
	title = BLANK_STRING
	picture = GFX_evt_hyperlane
	desc = magiks_decision_hyperlane_found_desc
	is_triggered_only = yes
	option = {
		name = LETS_START
		custom_tooltip = magiks_event_hyperlane_out_border_resource_fail
		allow = {
			space_owner = {
				has_resource = {
					type = energy
					amount >= 2500
				}
				has_resource = {
					type = manacite
					amount >= 150
				}
				has_resource = {
					type = influence
					amount >= 4
				}
			}
		}
		hidden_effect = {
			add_hyperlane = {
				from = From.solar_system
				to = This
			}
			space_owner = {
				add_resource = {
					energy = -1000
					manacite = -50
					influence = -1
				}
			}
			From = {
				if = {
					limit = { exists = space_owner }
					space_owner = {
						#log = "Space Owner: [This.GetName] PrevPrev: [PrevPrev.GetName] PrevPrevspace_owner: [PrevPrev.space_owner.GetName]"
						add_opinion_modifier = {
							who = PrevPrev.space_owner
							modifier = triggered_opinion_disapprove_of_hyperlane_manipulation
						}
					}
				}
			}
		} 
	}
	option = {
		name = NEVERMIND
	}
}
